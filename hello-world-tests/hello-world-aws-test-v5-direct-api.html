🚀 AWS Evidence - Direct API Test

CSP制限対応版：外部SDK不要

この方式の利点:

  • ✅ 外部スクリプト読み込み不要
  • ✅ CSPエラーなし
  • ✅ 企業環境完全対応
  • ✅ ブラウザ標準API使用

使用方法:

  1. AWS Management Console にログイン
  2. F12 → Consoleタブを開く
  3. 以下のコードを貼り付けて実行

📋 AWS STS認証テスト

console.log('AWS STS Direct API Test'); fetch('https://sts.ap-northeast-1.amazonaws.com/', { method: 'POST', headers: { 'Content-Type': 'application/x-www-form-urlencoded', 'Accept': 'application/json' }, body: 'Action=GetCallerIdentity&Version=2011-06-15', credentials: 'include' }) .then(response => { console.log('Response status:', response.status); return response.text(); }) .then(data => { console.log('AWS STS Response:'); console.log(data); if (data.includes('')) { const accountMatch = data.match(/(\d+)<\/Account>/); const arnMatch = data.match(/([^<]+)<\/Arn>/); const userIdMatch = data.match(/([^<]+)<\/UserId>/); if (accountMatch) console.log('Account:', accountMatch[1]); if (arnMatch) console.log('ARN:', arnMatch[1]); if (userIdMatch) console.log('User ID:', userIdMatch[1]); } }) .catch(error => { console.error('STS Error:', error); });

📋 S3バケット一覧テスト

console.log('S3 Direct API Test'); fetch('https://s3.ap-northeast-1.amazonaws.com/', { method: 'GET', credentials: 'include' }) .then(response => { console.log('S3 Response status:', response.status); return response.text(); }) .then(data => { console.log('S3 Response:'); console.log(data); if (data.includes('')) { const bucketMatches = data.match(/([^<]+)<\/Name>/g); if (bucketMatches) { console.log(`Found ${bucketMatches.length} buckets:`); bucketMatches.forEach((match, index) => { const bucketName = match.replace(/|<\/Name>/g, ''); console.log(`${index + 1}. ${bucketName}`); }); } } }) .catch(error => { console.error('S3 Error:', error); });

📋 CloudWatch Logs テスト

console.log('CloudWatch Logs Direct API Test'); const logsPayload = JSON.stringify({ "limit": 10 }); fetch('https://logs.ap-northeast-1.amazonaws.com/', { method: 'POST', headers: { 'Content-Type': 'application/x-amz-json-1.1', 'X-Amz-Target': 'Logs_20140328.DescribeLogGroups' }, body: logsPayload, credentials: 'include' }) .then(response => { console.log('CloudWatch Logs Response status:', response.status); return response.json(); }) .then(data => { console.log('CloudWatch Logs Response:'); console.log(data); if (data.logGroups) { console.log(`Found ${data.logGroups.length} log groups:`); data.logGroups.forEach((group, index) => { console.log(`${index + 1}. ${group.logGroupName}`); }); } }) .catch(error => { console.error('CloudWatch Logs Error:', error); });

📋 まとめテスト（全部実行）

async function runAllAWSTests() { console.log('=== AWS Evidence System - Complete Test ==='); try { console.log('\n1. STS Authentication Test...'); const stsResponse = await fetch('https://sts.ap-northeast-1.amazonaws.com/', { method: 'POST', headers: { 'Content-Type': 'application/x-www-form-urlencoded' }, body: 'Action=GetCallerIdentity&Version=2011-06-15', credentials: 'include' }); const stsData = await stsResponse.text(); console.log('STS Status:', stsResponse.status); if (stsData.includes('')) { console.log('✅ Authentication successful'); const account = stsData.match(/(\d+)<\/Account>/)?.[1]; console.log('Account ID:', account); } else { console.log('❌ Authentication failed'); } console.log('\n2. S3 Buckets Test...'); const s3Response = await fetch('https://s3.ap-northeast-1.amazonaws.com/', { method: 'GET', credentials: 'include' }); const s3Data = await s3Response.text(); console.log('S3 Status:', s3Response.status); const buckets = s3Data.match(/([^<]+)<\/Name>/g); if (buckets) { console.log(`✅ Found ${buckets.length} S3 buckets`); } else { console.log('⚠️ No S3 buckets or access denied'); } console.log('\n=== Test Complete ==='); } catch (error) { console.error('Test Error:', error); } } runAllAWSTests();
